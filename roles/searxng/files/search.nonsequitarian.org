
upstream searxng {
	server localhost:8080;
}

server {
	server_name search.nonsequitarian.org;

	auth_basic "Members Only";
	auth_basic_user_file /etc/nginx/htpasswd;

	location / {
		proxy_pass http://searxng;
		include proxy_params;
		proxy_set_header X-Forwarded-Host $server_name;
	}


    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/search.nonsequitarian.org/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/search.nonsequitarian.org/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

	access_log syslog:server=unix:/dev/log,tag=search;
	error_log syslog:server=unix:/dev/log,tag=search_error;
}


server {
    if ($host = search.nonsequitarian.org) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


	server_name search.nonsequitarian.org;
    listen 80;
    return 404; # managed by Certbot


}
