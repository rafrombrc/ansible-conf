upstream minio {
	server localhost:3200;
}

server {
        server_name minio.nonsequitarian.org;

        location / {
                proxy_pass http://minio;
                proxy_set_header   Host $http_host;
                proxy_set_header   X-Forwarded-For $remote_addr;
                proxy_set_header   X-Forwarded-Host $server_name;
		proxy_set_header   Upgrade $http_upgrade;
		proxy_set_header   Connection "upgrade";
	
		if ($request_method = 'OPTIONS') {
			add_header 'Access-Control-Allow-Origin' 'https://ente.nonsequitarian.org';
			add_header 'Access-Control-Allow-Methods' 'GET, POST, PUT, OPTIONS';
			add_header 'Access-Control-Allow-Headers' 'DNT,User-Agent,X-Requested-With,If-Modified-Since,Cache-Control,Content-Type,Range,X-Client-Package,X-Auth-Token,X-Client-Version';
		        add_header 'Access-Control-Max-Age' 1728000;
		       	add_header 'Content-Type' 'text/plain; charset=utf-8';
			add_header 'Content-Length' 0;
		        return 204;
		}
		if ($request_method = 'POST') {
			# add_header 'Access-Control-Allow-Origin' 'https://ente.nonsequitarian.org' always;
			add_header 'Access-Control-Expose-Headers' 'Content-Length,Content-Range' always;
		}
		if ($request_method = 'GET') {
			# add_header 'Access-Control-Allow-Origin' '*' always;
			add_header 'Access-Control-Expose-Headers' 'Content-Length,Content-Range' always;
		}
	}

	client_max_body_size 10000M;
	proxy_max_temp_file_size 0;
	proxy_buffering off;

	access_log /var/log/nginx/minio.nonsequitarian.org-access.log;
	error_log /var/log/nginx/minio.nonsequitarian.org-error.log;

    listen [::]:443; # manually changed
    listen 443 ssl; # managed by Certbot
    ssl_certificate /etc/letsencrypt/live/nonsequitarian.org/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/nonsequitarian.org/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot
}
